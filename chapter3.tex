\section{Transport Layer Services and Protocols (3.1)}
%=============================================================================

\subsection{The Role of the Transport Layer}

\begin{keybox}[The Transport Layer -- Overview]
The transport layer provides \textbf{logical communication} between \emph{application processes} on different hosts.
\begin{itemize}[noitemsep]
    \item Sender side: breaks application messages into \textbf{segments}, sends them to the network layer
    \item Receiver side: reassembles segments into messages, delivers to the application layer
    \item Two main protocols: \textbf{TCP} and \textbf{UDP}
\end{itemize}
\end{keybox}

\subsection{Transport Layer vs. Network Layer}

\begin{keybox}[Important Difference]
\begin{itemize}[noitemsep]
    \item \textbf{Network layer (IP):} Logical communication between \emph{hosts}
    \item \textbf{Transport layer (TCP/UDP):} Logical communication between \emph{processes}
    \item The transport layer extends the network layer's host-to-host delivery to process-to-process delivery
\end{itemize}
\end{keybox}

\begin{warnbox}[Household Analogy]
Think of two houses with 12 children in each house sending letters to each other:
\begin{itemize}[noitemsep]
    \item \textbf{Processes} = children
    \item \textbf{Application messages} = letters in envelopes
    \item \textbf{Hosts} = houses
    \item \textbf{Transport protocol} = Ann and Bill (the children who collect/sort mail)
    \item \textbf{Network layer protocol} = the postal service
\end{itemize}
Ann and Bill work within the house -- they cannot control what happens out in the postal system.
\end{warnbox}

\subsection{Internet Transport Protocols}

\begin{keybox}[TCP vs. UDP -- Overview]
\textbf{TCP (Transmission Control Protocol):}
\begin{itemize}[noitemsep]
    \item Reliable, ordered delivery (\emph{reliable, in-order delivery})
    \item Congestion control (\emph{congestion control})
    \item Flow control (\emph{flow control})
    \item Connection setup (\emph{connection setup})
\end{itemize}

\textbf{UDP (User Datagram Protocol):}
\begin{itemize}[noitemsep]
    \item Unreliable, unordered delivery (\emph{unreliable, unordered delivery})
    \item No congestion control, flow control, or connection setup
    \item ``Best effort'' -- no guarantee of delivery
\end{itemize}

\textbf{Services NOT provided:} Delay or bandwidth guarantees.
\end{keybox}

\section{Multiplexing and Demultiplexing (3.2)}
%=============================================================================

\subsection{Basic Concept}

\begin{keybox}[Multiplexing and Demultiplexing]
\textbf{Multiplexing at sender:}
\begin{itemize}[noitemsep]
    \item Handles data from multiple sockets
    \item Adds transport header (with port number) to identify socket
\end{itemize}

\textbf{Demultiplexing at receiver:}
\begin{itemize}[noitemsep]
    \item Uses header information (port number) to deliver segments to the correct socket
\end{itemize}
\end{keybox}

\subsection{How Demultiplexing Works}

\begin{itemize}[noitemsep]
    \item The host receives IP datagrams -- each datagram has source IP, destination IP, and one transport layer segment
    \item Each segment has a \textbf{source port number} and \textbf{destination port number} (16 bits each)
    \item The host uses IP addresses and port numbers to direct the segment to the correct socket
\end{itemize}

\subsection{Connectionless Demultiplexing (UDP)}

\begin{keybox}[UDP socket identified by 2-tuple]
UDP socket identified by: \textbf{(destination IP, destination port)}
\begin{itemize}[noitemsep]
    \item Segments with the \emph{same} destination IP and destination port, but \emph{different} source IP/port, are directed to the \textbf{same socket}
    \item Source port is used as a ``return address''
\end{itemize}
\end{keybox}

\subsection{Connection-Oriented Demultiplexing (TCP)}

\begin{keybox}[TCP socket identified by 4-tuple]
TCP socket identified by: \textbf{(source IP, source port, destination IP, destination port)}
\begin{itemize}[noitemsep]
    \item Receiver uses \emph{all four values} to direct the segment to the correct socket
    \item The server host can support many simultaneous TCP sockets -- each identified by its own 4-tuple
    \item Web servers have different sockets for each client (different source IP and/or source port)
\end{itemize}
\end{keybox}

\begin{warnbox}[Important Difference UDP vs. TCP demux]
\begin{itemize}[noitemsep]
    \item \textbf{UDP:} Only destination port determines which socket receives the segment
    \item \textbf{TCP:} Both source and destination (IP + port) determine socket -- two clients with different source IP/port get \emph{separate} sockets
\end{itemize}
\end{warnbox}

\section{Connectionless Transport: UDP (3.3)}
%=============================================================================

\subsection{UDP -- Overview}

\begin{keybox}[UDP -- User Datagram Protocol (RFC 768)]
\begin{itemize}[noitemsep]
    \item ``Bare bones'' transport protocol -- ``no frills''
    \item \textbf{Best effort}: Segments can be lost or delivered in wrong order
    \item \textbf{Connectionless:} No handshake between sender and receiver, each segment handled independently
\end{itemize}

\textbf{Why use UDP?}
\begin{itemize}[noitemsep]
    \item No connection setup (which adds delay)
    \item Simple: no state at sender/receiver
    \item Small header (8 bytes vs. 20 bytes for TCP)
    \item No congestion control -- can send as fast as desired / the application allows
\end{itemize}
\end{keybox}

\subsection{Use Cases for UDP}

\begin{itemize}[noitemsep]
    \item Multimedia streaming (tolerates loss, rate-sensitive)
    \item DNS (fast, short queries)
    \item SNMP (network management)
    \item HTTP/3 (uses QUIC over UDP)
\end{itemize}

\begin{warnbox}[UDP and Reliability]
If reliable transfer is necessary over UDP, it must be added at the \textbf{application layer}. Example: HTTP/3 / QUIC adds reliability and congestion control over UDP.
\end{warnbox}

\subsection{UDP Checksum}

\begin{keybox}[Checksum -- Error Detection]
\textbf{Goal:} Detect ``errors'' (e.g., flipped bits) in the transmitted segment.

\textbf{Sender:}
\begin{itemize}[noitemsep]
    \item Treats the segment contents (including header fields) as a sequence of 16-bit words
    \item Adds all 16-bit words using \textbf{1's complement addition}
    \item Takes the \textbf{1's complement} of the result = checksum
    \item Places the checksum in the UDP header's checksum field
\end{itemize}

\textbf{Receiver:}
\begin{itemize}[noitemsep]
    \item Computes the checksum of the received segment (including the checksum field)
    \item If the result is \emph{not} \texttt{1111111111111111} $\Rightarrow$ error detected
    \item If the result is all 1s $\Rightarrow$ no error detected (but there may still be errors!)
\end{itemize}
\end{keybox}

\begin{formulabox}[Checksum Example]
Add two 16-bit words:
\begin{align*}
&\texttt{1110011001100110} \\
+\ &\texttt{1101010101010101} \\
=\ &\texttt{1011101110111011} \quad \text{(with wraparound carry)} \\
\text{Checksum} =\ &\texttt{0100010001000100} \quad \text{(1's complement)}
\end{align*}
\end{formulabox}

\section{Principles of Reliable Data Transfer (3.4)}
%=============================================================================

\subsection{Problem Statement}

Reliable data transfer is one of the most important problems in networking. The challenge is to build a \textbf{reliable channel} over an \textbf{unreliable channel} (which can lose packets, flip bits, etc.).

\subsection{rdt1.0: Transfer over a Perfectly Reliable Channel}

\begin{keybox}[rdt1.0 -- Simplest Case]
\textbf{Assumption:} Underlying channel is \emph{perfectly reliable} (no bit errors, no packet loss).

\textbf{Sender:} Just send data down into the channel.\\
\textbf{Receiver:} Just read data from the channel.

No error handling necessary -- trivial!
\end{keybox}

\subsection{rdt2.0: Channel with Bit Errors}

\begin{keybox}[rdt2.0 -- Stop-and-Wait with ACK/NAK]
\textbf{Assumption:} The channel may have \emph{bit errors} (but no packet loss).

\textbf{New mechanisms:}
\begin{itemize}[noitemsep]
    \item \textbf{Checksum:} Detects bit errors
    \item \textbf{ACK (acknowledgement):} Receiver says ``OK, received correctly''
    \item \textbf{NAK (negative acknowledgement):} Receiver says ``error, send again''
    \item \textbf{Retransmission:} Sender retransmits on NAK
\end{itemize}
The sender sends one packet and \textbf{waits} for ACK/NAK before the next packet is sent (\emph{stop-and-wait}).
\end{keybox}

\begin{warnbox}[Fatal Problem with rdt2.0!]
What if ACK/NAK itself has bit errors? The sender does not know if the packet was received correctly!
\begin{itemize}[noitemsep]
    \item Cannot simply retransmit -- causes possible duplicates
    \item Solution $\rightarrow$ rdt2.1
\end{itemize}
\end{warnbox}

\subsection{rdt2.1: Sequence Numbers}

\begin{keybox}[rdt2.1 -- Solution: Sequence Numbers]
Sender adds a \textbf{sequence number} (0 or 1) to each packet:
\begin{itemize}[noitemsep]
    \item If ACK/NAK is corrupt $\rightarrow$ sender retransmits current packet
    \item Receiver discards duplicates using the sequence number
    \item Two sequence numbers (0 and 1) are sufficient for stop-and-wait
\end{itemize}
\end{keybox}

\subsection{rdt2.2: NAK-Free Protocol}

\begin{keybox}[rdt2.2 -- ACK Only (no NAK)]
Same functionality as rdt2.1, but \emph{without NAK}:
\begin{itemize}[noitemsep]
    \item Receiver sends \textbf{ACK with the sequence number} of the last correctly received packet
    \item Duplicate ACK at sender acts as NAK (means: ``the last packet was wrong'')
    \item Sender retransmits when it receives a duplicate ACK
\end{itemize}
\end{keybox}

\subsection{rdt3.0: Channel with Bit Errors AND Packet Loss}

\begin{keybox}[rdt3.0 -- Timer / Countdown Timer]
\textbf{New assumption:} The channel can also \emph{lose packets} (data or ACKs).

\textbf{New mechanism: Countdown timer}
\begin{itemize}[noitemsep]
    \item Sender waits a ``reasonable'' time for ACK
    \item Retransmits if no ACK is received before timeout
    \item If the packet (or ACK) is just delayed (not lost): retransmission causes a duplicate, but sequence numbers handle this
\end{itemize}
rdt3.0 is correct, but performance is poor on high-capacity links!
\end{keybox}

\subsection{Performance of rdt3.0 (Stop-and-Wait)}

\begin{formulabox}[Utilization for Stop-and-Wait]
\[
U_{\text{sender}} = \frac{L/R}{RTT + L/R}
\]
where:
\begin{itemize}[noitemsep]
    \item $L$ = packet size (bits)
    \item $R$ = link capacity (bps)
    \item $RTT$ = round-trip time
    \item $L/R$ = transmission time for one packet
\end{itemize}
\end{formulabox}

\begin{exambox}[Example: Poor Utilization]
Given: $L = 8000$ bits, $R = 1$ Gbps, $RTT = 30$ ms:
\[
U_{\text{sender}} = \frac{8000 / 10^9}{0{,}030 + 8000/10^9} = \frac{0{,}000008}{0{,}030008} = 0{,}00027 = 0{,}027\%
\]
Only 0.027\% utilization! The sender is active only $0{,}008$ ms out of $30{,}008$ ms.
\end{exambox}

\subsection{Pipelining: Increased Utilization}

\begin{keybox}[Pipelining -- Send Multiple Packets Without Waiting]
Solution to poor utilization: \textbf{pipelining} -- the sender allows multiple packets to be ``in flight'' (sent but not yet acknowledged).
\begin{itemize}[noitemsep]
    \item Requires a larger range of sequence numbers
    \item Buffering at sender and/or receiver
    \item Two main variants: \textbf{Go-Back-N} and \textbf{Selective Repeat}
\end{itemize}

\textbf{With pipelining (3 packets):}
\[
U_{\text{sender}} = \frac{3 \cdot L/R}{RTT + L/R}
\]
3x better utilization than stop-and-wait!
\end{keybox}

\subsection{Go-Back-N (GBN)}

\begin{keybox}[Go-Back-N Protocol]
\textbf{Sender:}
\begin{itemize}[noitemsep]
    \item ``Window'' of up to $N$ unacknowledged packets in the pipeline
    \item \textbf{Cumulative ACK:} ACK($n$) acknowledges all packets up to and including sequence number $n$
    \item Timer for the oldest unacknowledged packet
    \item On timeout: retransmit \emph{all} packets in the window (from the oldest unacknowledged)
\end{itemize}

\textbf{Receiver:}
\begin{itemize}[noitemsep]
    \item Sends ACK only for the highest correctly received packet \emph{in order}
    \item Can have only one expected sequence number
    \item Discards out-of-order packets (no buffering) -- re-sends ACK for last in-order packet
\end{itemize}
\end{keybox}

\subsection{Selective Repeat (SR)}

\begin{keybox}[Selective Repeat Protocol]
\textbf{Sender:}
\begin{itemize}[noitemsep]
    \item Window of up to $N$ unacknowledged packets
    \item \textbf{Individual ACK:} Each packet has its own ACK
    \item Timer for \emph{each} unacknowledged packet
    \item On timeout: retransmit only the \emph{specific} packet that timed out
\end{itemize}

\textbf{Receiver:}
\begin{itemize}[noitemsep]
    \item Acknowledges \emph{each} correctly received packet individually
    \item \textbf{Buffers} out-of-order received packets
    \item Delivers to the application when the ``gap'' is filled in
\end{itemize}
\end{keybox}

\begin{warnbox}[SR Dilemma: Window Size vs. Sequence Number Space]
The window size $N$ must be \textbf{at most half the sequence number space}:
\[
N \leq \frac{k}{2} \quad \text{where } k \text{ is the number of possible sequence numbers}
\]
Otherwise the receiver cannot distinguish between new packets and retransmissions!

\textbf{Example:} With sequence numbers $\{0, 1, 2, 3\}$ (4 possible), $N \leq 2$.
\end{warnbox}

\begin{tabular}{lcc}
\toprule
\textbf{Property} & \textbf{Go-Back-N} & \textbf{Selective Repeat} \\
\midrule
ACK type & Cumulative & Individual \\
Receiver buffering & No & Yes \\
On timeout & Retransmit all & Retransmit one \\
Timer & One (oldest) & Per packet \\
Complexity & Lower & Higher \\
Efficiency & Lower (many retransm.) & Higher \\
\bottomrule
\end{tabular}
\end{center}

\section{Connection-Oriented Transport: TCP (3.5)}
%=============================================================================

\subsection{TCP -- Overview}

\begin{keybox}[TCP -- Properties]
\begin{itemize}[noitemsep]
    \item \textbf{Point-to-point:} One sender, one receiver
    \item \textbf{Reliable, ordered byte stream:} No message boundaries
    \item \textbf{Full duplex:} Data can flow in both directions simultaneously (bi-directional data flow)
    \item \textbf{Cumulative ACK}
    \item \textbf{Pipelining:} Window-controlled segments (congestion \& flow control set window)
    \item \textbf{Connection-oriented:} Handshake before data exchange
    \item \textbf{Flow and congestion control:} Sender does not overwhelm receiver or network
\end{itemize}
\end{keybox}

\begin{keybox}[Important Fields in the TCP Header]
\begin{itemize}[noitemsep]
    \item \textbf{Sequence number:} Byte number of the first byte in the segment's data
    \item \textbf{ACK number:} Next byte the receiver expects (cumulative ACK)
    \item \textbf{Receive window (rwnd):} Number of bytes the receiver is willing to accept (flow control)
    \item \textbf{Flags:} SYN, FIN, ACK, RST, PSH, URG, CWR, ECE
    \item \textbf{Header length:} Variable due to options (normally 20 bytes without options)
\end{itemize}
\end{keybox}

\subsection{MSS and MTU}

\begin{formulabox}[MSS and MTU]
\begin{itemize}[noitemsep]
    \item \textbf{MTU (Maximum Transmission Unit):} Largest link layer frame (typically \textbf{1500 bytes} for Ethernet)
    \item \textbf{MSS (Maximum Segment Size):} Max application data in a segment
    \item $\text{MSS} = \text{MTU} - \text{IP header} - \text{TCP header} = 1500 - 20 - 20 = \textbf{1460 bytes}$
    \item MSS refers to application data, \emph{not} including the TCP header
\end{itemize}
\end{formulabox}

\subsection{TCP Sequence Numbers and ACK}

\begin{keybox}[Sequence Numbers and ACK in TCP]
\textbf{Sequence numbers:}
\begin{itemize}[noitemsep]
    \item TCP numbers \emph{bytes} (not segments!)
    \item The sequence number is the byte number of the first data byte in the segment
    \item Example: file of 500\,000 bytes, MSS = 1000 bytes $\Rightarrow$ segment 0 has seq=0, segment 1 has seq=1000, segment 2 has seq=2000, \ldots
\end{itemize}

\textbf{ACK numbers:}
\begin{itemize}[noitemsep]
    \item The ACK number is the sequence number of the \textbf{next byte the receiver expects}
    \item \textbf{Cumulative ACK:} Acknowledges all bytes up to ACK number $-$ 1
    \item Question: What does the receiver do with out-of-order packets? TCP spec says nothing -- up to implementation (in practice: buffer them)
\end{itemize}
\end{keybox}

\begin{exambox}[Example: Telnet Session]
Host A sends `C' (1 byte) to Host B, both start with seq=42 (A) and seq=79 (B):
\begin{enumerate}[noitemsep]
    \item A $\rightarrow$ B: Seq=42, ACK=79, data=`C'
    \item B $\rightarrow$ A: Seq=79, ACK=43, data=`C' (echo)
    \item A $\rightarrow$ B: Seq=43, ACK=80
\end{enumerate}
\end{exambox}

\subsection{TCP Round-Trip Time and Timeout}

\begin{formulabox}[RTT Estimation and Timeout]
\textbf{SampleRTT:} Measured time from segment sent to ACK received (only for segments sent once).

\textbf{Estimated RTT (exponentially weighted moving average -- EWMA):}
\[
\text{EstimatedRTT} = (1 - \alpha) \cdot \text{EstimatedRTT} + \alpha \cdot \text{SampleRTT}
\]
Typical: $\alpha = 0{,}125$ (i.e. $1/8$)

\textbf{Deviation measure (Safety margin):}
\[
\text{DevRTT} = (1 - \beta) \cdot \text{DevRTT} + \beta \cdot |\text{SampleRTT} - \text{EstimatedRTT}|
\]
Typical: $\beta = 0{,}25$ (i.e. $1/4$)

\textbf{Timeout interval:}
\[
\text{TimeoutInterval} = \text{EstimatedRTT} + 4 \cdot \text{DevRTT}
\]
\end{formulabox}

\begin{warnbox}[Important about SampleRTT]
SampleRTT is \textbf{not} measured for retransmitted segments (Karn's algorithm) -- one cannot know whether the ACK is for the original or the retransmission.
\end{warnbox}

\subsection{TCP Reliable Data Transfer}

\begin{keybox}[TCP Sender -- Simplified]
TCP uses \textbf{one retransmission timer} per connection (not per segment).

\textbf{Events at sender:}
\begin{enumerate}[noitemsep]
    \item \textbf{Data from application:} Create segment with sequence number, start timer if not already running
    \item \textbf{Timeout:} Retransmit the segment with the lowest sequence number that is not acknowledged. Restart timer
    \item \textbf{ACK received:} If ACK acknowledges previously unacknowledged segments -- update what is acknowledged. Restart timer if there are still unacknowledged segments
\end{enumerate}
\end{keybox}

\begin{keybox}[TCP Fast Retransmit]
If sender receives \textbf{3 duplicate ACKs} (i.e.\ 4 ACKs for the same segment): retransmit the segment \emph{before} timeout expires.

Rationale: Duplicate ACKs indicate that a segment has been lost, but subsequent segments have arrived.
\end{keybox}

\subsection{TCP Flow Control}

\begin{keybox}[Flow Control]
\textbf{Goal:} Sender should not overflow the receiver's buffer.

\textbf{Mechanism:}
\begin{itemize}[noitemsep]
    \item Receiver advertises available buffer space in the TCP header via the \textbf{rwnd} field (receive window)
    \item $\text{rwnd} = \text{RcvBuffer} - (\text{LastByteRcvd} - \text{LastByteRead})$
    \item Sender limits unacknowledged data: $\text{LastByteSent} - \text{LastByteAcked} \leq \text{rwnd}$
    \item Default RcvBuffer = 4096 bytes (can be increased via operating system)
\end{itemize}
\end{keybox}

\begin{warnbox}[What when rwnd = 0?]
When the receiver advertises $\text{rwnd} = 0$, the sender still sends small \textbf{probe segments} (1 byte) to get an updated rwnd value and avoid deadlock.
\end{warnbox}

\subsection{TCP Connection Management}

\subsubsection{Three-Way Handshake}

\begin{keybox}[TCP Connection Setup -- 3-Way Handshake]
\begin{enumerate}[noitemsep]
    \item \textbf{SYN:} Client sends SYN segment (SYN=1, chooses random sequence number \texttt{client\_isn})
    \item \textbf{SYN-ACK:} Server responds with SYNACK (SYN=1, ACK=1, chooses \texttt{server\_isn}, acks \texttt{client\_isn+1})
    \item \textbf{ACK:} Client sends ACK (ACK=1, acks \texttt{server\_isn+1}). Can contain data!
\end{enumerate}
\end{keybox}

A two-way handshake fails because:

\section{Principles of Congestion Control (3.6)}
%=============================================================================

\subsection{What is Congestion?}

\begin{keybox}[Congestion]
\textbf{Congestion} occurs when too many sources send too much data too fast, such that \textbf{the network} cannot handle it all.

\textbf{Symptoms:}
\begin{itemize}[noitemsep]
    \item Long queueing delays (queues in router buffers)
    \item Packet loss (buffer overflow in routers)
\end{itemize}

\textbf{Difference from flow control:}
\begin{itemize}[noitemsep]
    \item \textbf{Flow control:} Prevents sender from overwhelming the \emph{receiver}
    \item \textbf{Congestion control:} Prevents sender from overwhelming the \emph{network}
\end{itemize}
\end{keybox}

\subsection{Costs of Congestion}

\begin{warnbox}[Costs of Congestion]
\begin{itemize}[noitemsep]
    \item Queueing delays increase as arrival rate approaches link capacity
    \item Retransmissions due to packet loss -- ``wasted'' work
    \item Unnecessary retransmissions (premature timeout) -- the sender sends \emph{copies} of packets already delivered
    \item When a packet is dropped along a path, all upstream transmission capacity used on that packet is wasted
\end{itemize}
\end{warnbox}

\subsection{Approaches to Congestion Control}

\begin{keybox}[Two Main Approaches]
\textbf{1. End-to-end congestion control:}
\begin{itemize}[noitemsep]
    \item No explicit feedback from the network
    \item Congestion inferred from observed loss and delay
    \item TCP uses this approach
\end{itemize}

\textbf{2. Network-assisted congestion control:}
\begin{itemize}[noitemsep]
    \item Routers provide feedback to the sender
    \item Can be \emph{direct} (explicit rate) or \emph{indirect} (ECN bit in header)
    \item ECN (Explicit Congestion Notification): 2 bits in the IP header
\end{itemize}
\end{keybox}

\section{TCP Congestion Control (3.7)}
%=============================================================================

\subsection{Overview: CWND and RWND}

\begin{keybox}[The Two ``Brakes'' in TCP]
The TCP sender limits the transmission rate with two windows:
\begin{itemize}[noitemsep]
    \item \textbf{cwnd (Congestion Window):} Determined by the \emph{sender} based on observed congestion. \emph{Not} visible in the header (internal variable)
    \item \textbf{rwnd (Receive Window):} Determined by the \emph{receiver}, advertised in the TCP header
\end{itemize}

\textbf{The golden rule:}
\[
\text{LastByteSent} - \text{LastByteAcked} \leq \min(\text{cwnd}, \text{rwnd})
\]

\textbf{TCP send rate:}
\[
\text{rate} \approx \frac{\text{cwnd}}{\text{RTT}} \text{ bytes/sec}
\]
\end{keybox}

\begin{formulabox}[Bandwidth With and Without Congestion Control]
\textbf{Without} congestion control (pre-Tahoe):
\[
BW_i = \frac{\text{TCP Window}}{\text{RTT}} = \frac{\text{RWND} \cdot \text{MSS}}{\text{RTT}}
\]

\textbf{With} congestion control:
\[
BW_i = \frac{\text{TCP Window}}{\text{RTT}} = \frac{\min(\text{CWND}, \text{RWND}) \cdot \text{MSS}}{\text{RTT}}
\]
\end{formulabox}

\subsection{TCP Slow Start}

\begin{keybox}[Slow Start -- Exponential Growth]
At \textbf{connection start}:
\begin{itemize}[noitemsep]
    \item $\text{cwnd} = 1$ MSS
    \item Initial rate = MSS/RTT
    \item For \textbf{each ACK received}: $\text{cwnd} = \text{cwnd} + \text{MSS}$
    \item Results in \textbf{doubling} of cwnd per RTT (exponential growth)
\end{itemize}

\textbf{Example} (MSS = 500 bytes, RTT = 200 ms):\\
Initial rate = 500/0{,}2 = 20 kbps, but doubles every RTT.
\end{keybox}

\begin{exambox}[Slow Start Calculation]
If cwnd = 4500 and MSS = 1500:\\
3 segments are sent (4500/1500 = 3), and 3 ACKs arrive this RTT:
\begin{itemize}[noitemsep]
    \item ACK 1: cwnd = 4500 + 1500 = 6000
    \item ACK 2: cwnd = 6000 + 1500 = 7500
    \item ACK 3: cwnd = 7500 + 1500 = 9000
\end{itemize}
I.e.\ cwnd \textbf{doubled} from 4500 to 9000 in one RTT!
\end{exambox}

\subsection{Congestion Avoidance -- AIMD}

\begin{keybox}[Congestion Avoidance -- Linear Growth]
When cwnd $\geq$ ssthresh (slow start threshold):
\begin{itemize}[noitemsep]
    \item \textbf{Additive Increase:} For each ACK received:
    \[
    \text{cwnd} = \text{cwnd} + \text{MSS} \cdot \frac{\text{MSS}}{\text{cwnd}}
    \]
    Results in an increase of \textbf{1 MSS per RTT} (linear growth)
    \item \textbf{Multiplicative Decrease:} On packet loss: halve cwnd
\end{itemize}
Gives a ``sawtooth'' pattern (\emph{saw tooth behavior}) -- continually probing for available bandwidth.
\end{keybox}

\begin{exambox}[Congestion Avoidance Calculation]
If cwnd = 3000 and MSS = 1500 (in Congestion Avoidance):
\begin{itemize}[noitemsep]
    \item cwnd0 = 3000 (start of RTT)
    \item 3000/1500 = 2 segments sent $\Rightarrow$ 2 ACKs
    \item ACK 1: cwnd = 3000 + 1500$\cdot$(1500/3000) = 3000 + 750 = 3750
    \item ACK 2: cwnd = 3750 + 1500$\cdot$(1500/3750) = 3750 + 600 = 4350
\end{itemize}
I.e.\ cwnd increased by approx.\ \textbf{1 MSS} to the next RTT (linear growth). (Exactly 1 MSS at the start of the next RTT period.)
\end{exambox}

\subsection{Loss Events and ssthresh}

\begin{keybox}[Reaction to Loss -- ssthresh]
The variable \textbf{ssthresh} (slow start threshold) determines when to switch from Slow Start to Congestion Avoidance.

\textbf{On loss event:}
\begin{itemize}[noitemsep]
    \item $\text{ssthresh} = \text{cwnd} / 2$
\end{itemize}

\textbf{Two types of loss -- different responses:}

\textbf{1. Timeout:}
\begin{itemize}[noitemsep]
    \item ssthresh = cwnd/2
    \item cwnd = 1 MSS
    \item Go back to \textbf{Slow Start}
\end{itemize}

\textbf{2. Triple duplicate ACK (3 dup ACK):}
\begin{itemize}[noitemsep]
    \item ssthresh = cwnd/2
    \item cwnd = ssthresh + 3 MSS
    \item Enter \textbf{Fast Recovery}
\end{itemize}
\end{keybox}

\subsection{TCP Tahoe vs. TCP Reno}

\begin{keybox}[TCP Tahoe vs. TCP Reno]
\textbf{TCP Tahoe (1988):}
\begin{itemize}[noitemsep]
    \item On \emph{all} loss events (both timeout and 3 dup ACK): cwnd = 1 MSS, go to Slow Start
\end{itemize}

\textbf{TCP Reno (1997, RFC 2001/2581):}
\begin{itemize}[noitemsep]
    \item On \textbf{timeout}: cwnd = 1 MSS, go to Slow Start (like Tahoe)
    \item On \textbf{3 dup ACK}: cwnd = ssthresh + 3, go to Fast Recovery (milder reaction)
    \item Fast Recovery gives better performance -- packet loss via dup-ACK means the network is still delivering some packets
\end{itemize}
\end{keybox}

\subsection{TCP CUBIC}

\begin{keybox}[TCP CUBIC -- Modern Congestion Control]
TCP CUBIC is the \textbf{default in Linux} and the most widely used TCP variant for web servers.

\textbf{Motivation:} Classic AIMD (linear increase) is too slow when bandwidth is high.

\textbf{Idea:}
\begin{itemize}[noitemsep]
    \item $W_{\max}$ = send rate when congestion was last detected
    \item After halving cwnd: increase quickly back toward $W_{\max}$, but slow down near $W_{\max}$
    \item $K$ = point in time when the window reaches $W_{\max}$ (configurable)
    \item Increase of $W$ as a function of the \textbf{cube} of the distance between current time and $K$
    \item Larger increases when far from $K$, smaller increases near $K$ (cautious)
\end{itemize}

\textbf{Advantages:}
\begin{itemize}[noitemsep]
    \item Independent of RTT (uses wall-clock time)
    \item Fair across different RTTs
    \item Higher throughput than classic TCP on high-speed links
\end{itemize}
\end{keybox}

\subsection{TCP Fairness}

\begin{keybox}[TCP Fairness]
\textbf{Goal:} If $K$ TCP sessions share a bottleneck link with capacity $R$, each should get an average rate of $R/K$.

\textbf{Why AIMD gives fairness:}
\begin{itemize}[noitemsep]
    \item Additive Increase: both sessions increase with slope 1 (parallel to the equal-share line)
    \item Multiplicative Decrease: both halve (moves toward the equal-share line)
    \item Over time the sessions converge toward an equal share of the bandwidth
\end{itemize}

\textbf{Note:} This is a very theoretical result -- in practice, RTT, number of connections per application, and UDP traffic (which has no congestion control) affect fairness.
\end{keybox}

\subsection{The Bottleneck Link}

\begin{warnbox}[Bottleneck Link and TCP Behavior]
TCP (both classic and CUBIC) increases the send rate until packet loss occurs at a router's output -- the \textbf{bottleneck link}.

\textbf{Key insights:}
\begin{itemize}[noitemsep]
    \item Increased TCP send rate \textbf{will not} increase end-to-end throughput when the bottleneck is saturated
    \item Increased send rate \textbf{will} increase measured RTT (due to queues)
    \item Goal: ``Keep the end-to-end pipe exactly full, but not fuller''
\end{itemize}
\end{warnbox}

\textbf{QUIC} is an application layer protocol built \textbf{over UDP} that provides:
\begin{itemize}[noitemsep]
    \item Reliable data transfer
    \item Congestion control
    \item Authentication and encryption
    \item Connection setup in \textbf{1 RTT} (vs. TCP+TLS = 2 serial handshakes)

    \item Streams are \textbf{independent} -- avoids TCP's head-of-line blocking
    \item Separate reliable transfer and security per stream
    \item Shared congestion control across streams
\end{itemize}

\section{Summary}
%=============================================================================

\begin{keybox}[Chapter 3 -- Key Points]
\begin{itemize}[noitemsep]
    \item \textbf{Transport layer} provides logical communication between \emph{processes} (not hosts)
    \item \textbf{Multiplexing/demultiplexing:} Port numbers direct to the correct socket (UDP: 2-tuple, TCP: 4-tuple)
    \item \textbf{UDP:} Simple, connectionless, ``best effort'' -- just mux/demux + checksum
    \item \textbf{Reliable data transfer:} Checksum, sequence numbers, ACK, timer, retransmission
    \item \textbf{Pipelining:} GBN (cumulative ACK, retransmit all) vs. SR (individual ACK, buffer out-of-order)
    \item \textbf{TCP:} Reliable, ordered byte stream with flow control, congestion control and connection management
    \item \textbf{TCP congestion control:} Slow Start (exponential), Congestion Avoidance (AIMD), Fast Recovery
    \item \textbf{TCP CUBIC:} Cubic window increase, default in Linux, RTT-independent
    \item \textbf{QUIC/HTTP/3:} Modern alternative over UDP with 1-RTT connection and independent streams
\end{itemize}
\end{keybox}

\section*{Formula Summary}
%=============================================================================

\begin{formulabox}[All Important Formulas -- Chapter 3]

\textbf{UDP checksum:} 1's complement of the sum of all 16-bit words.

\textbf{Stop-and-wait utilization:}
\[
U_{\text{sender}} = \frac{L/R}{RTT + L/R}
\]

\textbf{Pipelining utilization ($n$ packets):}
\[
U_{\text{sender}} = \frac{n \cdot L/R}{RTT + L/R}
\]

\textbf{MSS:}
\[
\text{MSS} = \text{MTU} - \text{IP header} - \text{TCP header} = 1500 - 20 - 20 = 1460 \text{ bytes}
\]

\textbf{RTT estimation:}
\begin{align*}
\text{EstimatedRTT} &= (1 - \alpha) \cdot \text{EstimatedRTT} + \alpha \cdot \text{SampleRTT} & (\alpha = 0{,}125) \\
\text{DevRTT} &= (1 - \beta) \cdot \text{DevRTT} + \beta \cdot |\text{SampleRTT} - \text{EstimatedRTT}| & (\beta = 0{,}25)
\end{align*}

\textbf{Timeout:}
\[
\text{TimeoutInterval} = \text{EstimatedRTT} + 4 \cdot \text{DevRTT}
\]

\textbf{TCP flow control:}
\[
\text{rwnd} = \text{RcvBuffer} - (\text{LastByteRcvd} - \text{LastByteRead})
\]

\textbf{TCP send limit:}
\[
\text{LastByteSent} - \text{LastByteAcked} \leq \min(\text{cwnd}, \text{rwnd})
\]

\textbf{TCP send rate:}
\[
\text{rate} \approx \frac{\min(\text{cwnd}, \text{rwnd})}{\text{RTT}} \text{ bytes/sec}
\]

\textbf{Slow Start (per ACK):} cwnd = cwnd + MSS \quad (doubling per RTT)

\textbf{Congestion Avoidance (per ACK):} cwnd = cwnd + MSS $\cdot$ (MSS/cwnd) \quad (+1 MSS per RTT)

\textbf{On timeout:} ssthresh = cwnd/2, \quad cwnd = 1 MSS

\textbf{On 3 dup ACK:} ssthresh = cwnd/2, \quad cwnd = ssthresh + 3 MSS

\textbf{TCP fairness:} $K$ sessions, capacity $R$ $\Rightarrow$ each gets $R/K$

\textbf{SR window size:} $N \leq k/2$ where $k$ = number of sequence numbers

\end{formulabox}
