\section{What is the Internet?}
%=============================================================================

The Internet can be described from two perspectives: \textbf{``nuts and bolts''} (the components) and \textbf{``service''} (the services).

\subsection{``Nuts and Bolts'' -- The Component Perspective}

\begin{keybox}[The Internet is a ``network of networks'']
The Internet consists of billions of connected devices that communicate via \textbf{packet switches} (routers and switches) over \textbf{communication links} (fiber, copper, radio, satellite).
\end{keybox}

\textbf{Key components:}
\begin{itemize}[noitemsep]
    \item \textbf{Hosts / end systems} -- devices that run network applications (PCs, mobiles, servers, IoT)
    \item \textbf{Packet switches} -- forward packets: \emph{routers} (network layer) and \emph{switches} (link layer)
    \item \textbf{Communication links} -- fiber, copper, radio, satellite. Transmission rate = \emph{bandwidth}
    \item \textbf{Networks} -- collection of devices, routers, and links managed by an organization
    \item \textbf{ISP} (Internet Service Provider) -- provides access to the Internet
\end{itemize}

\begin{keybox}[Internet Standards]
\begin{itemize}[noitemsep]
    \item \textbf{RFC} (Request for Comments) -- documents that describe the standards
    \item \textbf{IETF} (Internet Engineering Task Force) -- develops the standards
\end{itemize}
\end{keybox}

\subsection{``Service'' -- The Service Perspective}

The Internet is an \textbf{infrastructure that delivers services to applications}: web, video, e-mail, games, e-commerce, social media, IoT, etc.

The Internet offers a \textbf{programming interface} (socket API) to distributed applications -- ``hooks'' that allow apps to connect and use transport services.

\section{What is a protocol?}
%=============================================================================

\begin{keybox}[Definition: Protocol]
A \textbf{protocol} defines the \textbf{format}, the \textbf{order} of messages sent and received between network devices, and the \textbf{actions} taken when messages are sent or received.

\subsection{Hosts send packets}

\begin{formulabox}[Transmission Delay]
A host divides application messages into \textbf{packets} of length $L$ bits and sends them at transmission rate $R$ bits/sec:
\[

\]
\end{formulabox}

\section{The Network Core}

The network core is a network of interconnected routers -- a ``network of networks''.

\subsection{Packet Switching}

\begin{keybox}[Store-and-forward]
The entire packet must arrive at a router before it can be forwarded to the next link.\\
Delay for one packet over one link: $d = L/R$
\end{keybox}

Two key functions:
\begin{enumerate}[noitemsep]
    \item \textbf{Forwarding}: Moving a packet from an input link to the correct output link
    \item \textbf{Routing}: Determining the route the packet should take from source to destination
\end{enumerate}

\subsection{Four types of delay}

\begin{formulabox}[Total nodal delay]
\[
d_{\text{nodal}} = d_{\text{proc}} + d_{\text{queue}} + d_{\text{trans}} + d_{\text{prop}}

\begin{center}
\begin{tabular}{llll}
\toprule
\textbf{Delay} & \textbf{Formula} & \textbf{Depends on} & \textbf{Typical} \\
\midrule
$d_{\text{proc}}$ -- Processing & -- & Bit error check, choosing link & $< 1$ ms \\
$d_{\text{queue}}$ -- Queuing & -- & Traffic intensity & varies \\
$d_{\text{trans}}$ -- Transmission & $L/R$ & Packet size, link rate & $\mu$s--ms \\

\bottomrule
\end{tabular}
\end{center}

\begin{warnbox}[Important: Transmission delay $\neq$ Propagation delay!]
\begin{itemize}[noitemsep]
    \item $d_{\text{trans}} = L/R$ -- time to \textbf{push all bits out onto the link}
    \item $d_{\text{prop}} = d/s$ -- time for the signal to \textbf{travel through the link}
    \item $s \approx 2 \times 10^8$ m/s in fiber (speed of light / refractive index 1.47)
\end{itemize}
\textbf{Example:} Trondheim--Oslo (540 km), fiber:\\
$d_{\text{prop}} = 540\,000 / 200\,000\,000 = 2{,}7$ ms\\
Packet 1500 bytes, link 1 Gbps:\\
$d_{\text{trans}} = 12\,000 / 1\,000\,000\,000 = 0{,}012$ ms
\end{warnbox}

\subsection{Traffic intensity and queuing delay}

\begin{formulabox}[Traffic Intensity]
\[
\text{Traffic intensity} = \frac{L \cdot a}{R}
\]
where $L$ = packet size (bits), $a$ = average arrival rate (packets/s), $R$ = link rate (bps).
\end{formulabox}

\subsection{Packet Loss}

When a packet arrives at a router with a \textbf{full buffer}, the packet is \textbf{dropped} (lost). Lost packets may be retransmitted by the previous node, the source end system, or not at all.

\subsection{Throughput}

\begin{keybox}[Throughput and bottleneck]
\textbf{Throughput}: Rate (bits/time unit) at which bits are actually transferred from sender to receiver.
\begin{itemize}[noitemsep]
    \item \textbf{Instantaneous throughput}: Rate at a given point in time
    \item \textbf{Average throughput}: Rate over a longer period
    \item \textbf{Bottleneck link}: The link with the lowest capacity determines end-to-end throughput
\end{itemize}
\[
\text{Throughput} = \min(R_s, R_c)
\]
With 10 simultaneous connections sharing a backbone $R$:
\[
\text{Per-connection throughput} = \min(R_c, R_s, R/10)
\]
\end{keybox}

\subsection{Traceroute}

\textbf{Traceroute} measures delay along the entire route from source to destination:
\begin{itemize}[noitemsep]
    \item Sends 3 packets to each router $i$ along the way (with TTL = $i$)
    \item Router $i$ returns the packet, and the sender measures the time
    \item Also shows packet loss (``* * *'' = no response)
\end{itemize}

\section{Summary -- Key Formulas}
%=============================================================================

\begin{formulabox}[Formulas you must know]
\begin{align}
d_{\text{trans}} &= \frac{L}{R} & &\text{Transmission delay (packet size / link rate)} \\[6pt]
d_{\text{prop}} &= \frac{d}{s} & &\text{Propagation delay (distance / signal speed)} \\[6pt]
d_{\text{nodal}} &= d_{\text{proc}} + d_{\text{queue}} + d_{\text{trans}} + d_{\text{prop}} & &\text{Total delay per node} \\[6pt]
\text{Traffic intensity} &= \frac{L \cdot a}{R} & &\text{Arrival rate vs.\ service rate} \\[6pt]
\text{Throughput} &= \min(R_s, R_c) & &\text{Determined by the bottleneck}
\end{align}

\textbf{Values to remember:}
\begin{itemize}[noitemsep]
    \item Signal speed in fiber: $s \approx 2 \times 10^8$ m/s
    \item Speed of light: $c = 3 \times 10^8$ m/s (fiber has refractive index $\approx 1{,}47$)
    \item 1 byte = 8 bits
\end{itemize}
\end{formulabox}
