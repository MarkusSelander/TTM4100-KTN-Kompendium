\section{Principles of Network Applications (2.1)}

\subsection{Application Architectures}

Two dominant architectures for network applications:

\begin{keybox}[Client-Server vs. Peer-to-Peer (P2P)]
\textbf{Client-Server:}
\begin{itemize}[noitemsep]
    \item The server is \textbf{always on} with a fixed (permanent) IP address
    \item Clients contact the server, can have dynamic IP
    \item Clients do \textbf{not communicate directly} with each other
    \item Examples: HTTP, IMAP, FTP
    \item Scaling: multiple servers in data centers
\end{itemize}

\textbf{Peer-to-Peer (P2P):}
\begin{itemize}[noitemsep]
    \item \textbf{No} dedicated server -- end systems communicate directly
    \item Peers provide services to each other
    \item \textbf{Self-scaling}: new peers bring both capacity and demand
    \item Challenge: peers are intermittently connected, change IP
    \item Examples: BitTorrent, Skype, KanKan
\end{itemize}
\end{keybox}

\subsection{Processes and Sockets}

\begin{keybox}[Processes and Communication]
\begin{itemize}[noitemsep]
    \item A \textbf{process} is a program running on a host
    \item On the \textbf{same} host: inter-process communication (IPC)
    \item On \textbf{different} hosts: exchange of \textbf{messages}
    \item \textbf{Client process}: initiates communication
    \item \textbf{Server process}: waits to be contacted
\end{itemize}
\end{keybox}

\begin{keybox}[Socket -- The ``Door'' between Application and Transport]
A \textbf{socket} is the interface between the application layer and the transport layer. The process sends/receives messages through its socket.
\begin{itemize}[noitemsep]
    \item Two sockets involved: one on each side
    \item Application developer controls everything \textbf{above} the socket
    \item OS controls everything \textbf{below} the socket (transport, network, link, physical)
\end{itemize}
\end{keybox}

\subsection{Addressing Processes}

\begin{warnbox}[IP address alone is not enough!]
A process is uniquely identified by \textbf{IP address + port number}.
\begin{itemize}[noitemsep]
    \item The IP address identifies the \textbf{host}
    \item The port number identifies the \textbf{process} on the host
    \item Well-known ports: HTTP = \textbf{80}, Mail (SMTP) = \textbf{25}, DNS = \textbf{53}
\end{itemize}
\end{warnbox}

\subsection{Application-Layer Protocols}

An application-layer protocol defines:
\begin{enumerate}[noitemsep]
    \item \textbf{Message types}: request and response
    \item \textbf{Syntax}: the fields in the message and their format
    \item \textbf{Semantics}: the meaning of the fields
    \item \textbf{Rules}: when processes send/respond
\end{enumerate}

\textbf{Open protocols}: defined in RFCs, available to everyone (HTTP, SMTP).\\
\textbf{Proprietary protocols}: private (Skype, Zoom).

\section{Web and HTTP (2.2)}
%=============================================================================

\subsection{HTTP Basics}

\begin{keybox}[HTTP -- HyperText Transfer Protocol]
\begin{itemize}[noitemsep]
    \item A \textbf{web page} consists of a base HTML file and referenced objects (images, scripts, etc.)
    \item HTTP follows the \textbf{client-server} model
    \item Uses \textbf{TCP} (port \textbf{80})
    \item HTTP is \textbf{stateless} -- the server does not store information about previous requests
\end{itemize}
\end{keybox}

\subsection{Persistent vs. Non-persistent HTTP}

\begin{warnbox}[Non-persistent HTTP: Time per object]
Each object requires: \textbf{2 RTT + file transfer time}
\begin{itemize}[noitemsep]
    \item 1 RTT for TCP connection
    \item 1 RTT for HTTP request/response
    \item Problems: high RTT cost, OS overhead for many TCP connections
\end{itemize}
\end{warnbox}

\begin{formulabox}[Response time for Non-persistent HTTP]
$$T_{\text{non-pers}} = 2 \cdot RTT + T_{\text{fil}}$$
For a web page with base HTML + $n$ objects:
$$T_{\text{total}} = (n+1) \times (2 \cdot RTT + T_{\text{fil}})$$
\end{formulabox}

\subsection{HTTP Messages}

\textbf{Two types:} Request and Response.

\begin{keybox}[HTTP Request -- important methods]
\begin{tabular}{ll}
\toprule
\textbf{Method} & \textbf{Description} \\
\midrule
\texttt{GET} & Retrieves a resource; data in URL after ``?'' \\
\texttt{POST} & Sends data in the message body (entity body) \\
\texttt{HEAD} & Like GET, but returns headers only \\
\texttt{PUT} & Uploads / replaces a file on the server \\
\bottomrule
\end{tabular}
\end{keybox}

\begin{keybox}[HTTP Response -- status codes]
\begin{tabular}{ll}
\toprule
\textbf{Code} & \textbf{Meaning} \\
\midrule
\texttt{200 OK} & The request was successful \\
\texttt{301 Moved Permanently} & The object has a new permanent URL \\
\texttt{400 Bad Request} & The server did not understand the message \\
\texttt{404 Not Found} & The document does not exist on the server \\
\bottomrule
\end{tabular}
\end{keybox}

\subsection{Cookies}

\begin{keybox}[Cookies -- State information in stateless HTTP]
Cookies are used to maintain state across HTTP requests.

\textbf{Four components:}
\begin{enumerate}[noitemsep]
    \item Cookie header in HTTP \textbf{response} (Set-Cookie)
    \item Cookie header in HTTP \textbf{request} (Cookie)
    \item Cookie file on the \textbf{client} (stored by the browser)
    \item \textbf{Database} on the server
\end{enumerate}

Use cases: authorization, shopping carts, recommendations, session state.
\end{keybox}

\textit{Interaction between client and server showing Set-Cookie in response, Cookie in subsequent requests, and backend database.}
\end{imagebox}

\subsection{Web Caching (Proxy Server)}

\begin{keybox}[Web Cache -- Goal: satisfy requests without origin server]
\begin{itemize}[noitemsep]
    \item The browser sends \textbf{all} HTTP requests to the cache
    \item Object found? $\rightarrow$ cache returns it directly
    \item Not found? $\rightarrow$ cache fetches from origin server, caches it, returns it
    \item The cache acts both as a \textbf{server} (for the client) and \textbf{client} (for the origin server)
\end{itemize}
\end{keybox}

\textbf{Benefits of caching:}
\begin{enumerate}[noitemsep]
    \item Reduces response time for the client
    \item Reduces traffic on the institution's access link
    \item Cheaper than upgrading the access link
\end{enumerate}

\begin{keybox}[Conditional GET -- Avoid unnecessary transfer]
\begin{itemize}[noitemsep]
    \item Request header: \texttt{If-Modified-Since: <date>}
    \item Response if unchanged: \texttt{304 Not Modified} (no data sent)
    \item Response if changed: \texttt{200 OK} + updated object
\end{itemize}
\end{keybox}

\subsection{HTTP/2 and HTTP/3}

\begin{keybox}[HTTP/2 -- Reduce latency]
\begin{itemize}[noitemsep]
    \item Goal: reduce delay in multi-object requests
    \item \textbf{Priority-based transmission}: important objects first
    \item \textbf{Server push}: the server can send objects the client has not requested
    \item \textbf{Framing}: objects are divided into frames that are \textbf{interleaved} $\rightarrow$ reduces HOL-blocking
\end{itemize}
\end{keybox}

\begin{warnbox}[HOL-blocking (Head-of-Line)]
In HTTP/1.1, objects must be sent sequentially. A large object blocks small objects behind it in the queue. HTTP/2 solves this with frame interleaving.
\end{warnbox}

\begin{keybox}[HTTP/3 -- Solves TCP limitations]
\begin{itemize}[noitemsep]
    \item Uses \textbf{UDP} (via QUIC) instead of TCP
    \item Per-object error and congestion control
    \item Built-in security
    \item Solves the problem of \textit{packet loss on TCP affecting all objects}
\end{itemize}
\end{keybox}

\begin{enumerate}[noitemsep]

\subsection{The SMTP Protocol}

\begin{keybox}[SMTP -- Simple Mail Transfer Protocol]
\begin{itemize}[noitemsep]
    \item Port \textbf{25}, uses \textbf{TCP}
    \item \textbf{Push}-based (sender pushes data to receiver)
    \item Three phases: \textbf{handshaking}, \textbf{transfer}, \textbf{closure}
    \item Command/response interaction (ASCII text + status codes)
    \item Requires \textbf{7-bit ASCII} for the message body
    \item End of message: \texttt{CRLF.CRLF}
\end{itemize}
\end{keybox}

\begin{warnbox}[Differences between SMTP and HTTP]
\begin{tabular}{lll}
\toprule
& \textbf{SMTP} & \textbf{HTTP} \\
\midrule
Direction & \textbf{Push} (sender initiates) & \textbf{Pull} (receiver initiates) \\
Objects & Multipart in \textit{one} message & Each object separately \\
Encoding & 7-bit ASCII & Binary data allowed \\
Port & 25 & 80 \\
\bottomrule
\end{tabular}
\end{warnbox}

\subsection{Email Format and IMAP}

\begin{keybox}[Message Format]
\begin{itemize}[noitemsep]
    \item \textbf{Header}: To, From, Subject (part of the message content)
    \item \textbf{Body}: the actual message text
    \item Do not confuse with SMTP commands (MAIL FROM, RCPT TO) -- they are part of the protocol, not the message
\end{itemize}
\end{keybox}

\begin{keybox}[IMAP -- Internet Message Access Protocol]
\begin{itemize}[noitemsep]
    \item Used to \textbf{retrieve, organize, and delete} messages stored on the server
    \item Gmail/Hotmail use a web interface with SMTP (sending) and IMAP (retrieval)
\end{itemize}
\end{keybox}

\textbf{The steps when A sends an email to B:}
\begin{enumerate}[noitemsep]
    \item A composes email in their UA
    \item UA sends to A's mail server
    \item A's mail server opens TCP (port 25) to B's mail server
    \item The message is sent via SMTP
    \item The message is stored in B's mailbox
    \item B reads the message via their UA (IMAP)
\end{enumerate}

\section{DNS -- Domain Name System (2.4)}

\subsection{What is DNS?}

\begin{keybox}[DNS -- The Internet's Directory]
\begin{itemize}[noitemsep]
    \item Translates \textbf{hostnames} (e.g. \texttt{www.google.com}) to \textbf{IP addresses}
    \item Implemented as a \textbf{distributed, hierarchical database}
    \item Is an \textbf{application-layer protocol} -- hosts and DNS servers communicate directly to resolve names
\end{itemize}
\end{keybox}

\textbf{DNS services:}
\begin{enumerate}[noitemsep]
    \item Hostname $\rightarrow$ IP translation
    \item Host aliasing (canonical name vs. alias)
    \item Mail server aliasing
    \item Load distribution (load distribution)
\end{enumerate}

\begin{warnbox}[Why is DNS distributed?]
A centralized DNS would create: single point of failure, too much traffic, long distance, maintenance problems. DNS is ``highly distributed'' -- millions of organizations manage their own DNS records.
\end{warnbox}

\section{P2P File Distribution (2.5)}
%=============================================================================

\subsection{Client-Server vs. P2P Distribution}

\begin{keybox}[Distribution time -- The core question]
How long does it take to distribute a file (size $F$) from \textit{one} server to $N$ peers?

Variables:
\begin{itemize}[noitemsep]
    \item $u_s$ = server upload capacity
    \item $d_i$ = peer $i$'s download capacity
    \item $u_i$ = peer $i$'s upload capacity
    \item $d_{\min}$ = minimum download rate among all peers
\end{itemize}
\end{keybox}

\begin{formulabox}[Client-Server distribution time]
$$D_{CS} \geq \max\left\{\frac{NF}{u_s},\ \frac{F}{d_{\min}}\right\}$$
\begin{itemize}[noitemsep]
    \item $NF/u_s$: the server must send $N$ copies sequentially
    \item $F/d_{\min}$: the slowest client determines the minimum time
    \item \textbf{Grows linearly with $N$}
\end{itemize}
\end{formulabox}

\begin{formulabox}[P2P distribution time]
$$D_{P2P} \geq \max\left\{\frac{F}{u_s},\ \frac{F}{d_{\min}},\ \frac{NF}{u_s + \sum u_i}\right\}$$
\begin{itemize}[noitemsep]
    \item $F/u_s$: the server must upload \textit{at least} one copy
    \item $F/d_{\min}$: the slowest client cannot receive the file faster
    \item $NF/(u_s + \sum u_i)$: total load shared by \textbf{all} upload capacities
    \item \textbf{Self-scaling}: $\sum u_i$ grows with $N$!
\end{itemize}
\end{formulabox}

\begin{center}
\begin{tikzpicture}[>=Stealth]
    % Axes
    \draw[->] (0,0) -- (7.5,0) node[right, font=\small] {$N$};
    \draw[->] (0,0) -- (0,5) node[above, font=\small] {Min. distribution time};

    % Client-server (linear growth)
    \draw[thick, black, fill=black] plot[smooth, domain=0.5:7] (\x, {0.1*\x*\x/1.5 + 0.3});
    \node[font=\footnotesize, right] at (6.5,3.5) {Client-Server};

    % P2P (sub-linear, flattens)
    \draw[thick, blue] plot[smooth, domain=0.5:7] (\x, {0.3 + 0.7*ln(\x+1)/ln(8)});
    \node[font=\footnotesize, right, text=blue] at (6.5,1.5) {P2P};

    % Labels
    \foreach \x in {5,10,15,20,25,30} {
        \pgfmathsetmacro{\pos}{\x/5}
        \node[font=\tiny, below] at (\pos,0) {\x};
    }
\end{tikzpicture}
\end{center}

\begin{warnbox}[{{Example: $u_s = 10u$, $F/u = 1$ hour, $d_{\min} \geq u_s$}}]
\textbf{Client-Server:} $D_{CS} = NF/u_s = NF/(10u) = N/10$ (linear)\\
\textbf{P2P:} $D_{P2P} = NF/(u_s + Nu) = NF/((10+N)u) = N/(10+N)$ (flattens out!)

When $N=30$: CS takes $3$ hours, P2P takes $0.75$ hours.
\end{warnbox}

\section{Video Streaming and CDN (2.6)}
%=============================================================================

\subsection{Multimedia: Video}

\begin{keybox}[Video and encoding]
\begin{itemize}[noitemsep]
    \item Video = sequence of images displayed at a constant rate (e.g. 24 fps)
    \item Digital image = array of pixels, each represented with bits
    \item \textbf{Encoding} exploits redundancy to reduce bits:
    \begin{itemize}[noitemsep]
        \item \textbf{Spatial coding}: redundancy \textit{within} one image (e.g. many identical pixels)
        \item \textbf{Temporal coding}: redundancy \textit{between} images (send only differences)
    \end{itemize}
\end{itemize}
\end{keybox}

\begin{keybox}[CBR vs. VBR]
\begin{tabular}{lll}
\toprule
& \textbf{CBR} & \textbf{VBR} \\
\midrule
Full form & Constant Bit Rate & Variable Bit Rate \\
Encoding & Fixed rate & Rate changes with content \\
\bottomrule
\end{tabular}

\vspace{0.3em}
Examples: MPEG1 (1.5 Mbps), MPEG2 (3--6 Mbps), MPEG4 (64 Kbps -- 12 Mbps)
\end{keybox}

\subsection{Streaming of Stored Video}

\textbf{Challenges:}
\begin{itemize}[noitemsep]
    \item Bandwidth varies over time (congestion)
    \item Packet loss and delay cause poor quality
\end{itemize}

\begin{warnbox}[Continuous Playout Constraint]
Once playback starts, it must match the original timing. But network delays vary (\textbf{jitter}), so the client needs a \textbf{buffer} and \textbf{playout delay} to compensate.
\end{warnbox}

\subsection{DASH -- Dynamic Adaptive Streaming over HTTP}

\begin{keybox}[DASH]
\textbf{Server:}
\begin{itemize}[noitemsep]
    \item Divides video into \textbf{chunks} (segments)
    \item Each chunk encoded at \textbf{multiple quality levels} (bit rates)
    \item \textbf{Manifest file}: list of URLs for different chunks/qualities
\end{itemize}

\textbf{Client (``intelligence'' at client):}
\begin{itemize}[noitemsep]
    \item Periodically measures server-to-client bandwidth
    \item Decides \textbf{when} to fetch a chunk (avoid buffer starvation/overflow)
    \item Decides \textbf{which bit rate} (higher quality when more bandwidth)
    \item Decides \textbf{where} to fetch chunk from (nearest server)
\end{itemize}

\textbf{Streaming video = encoding + DASH + playout buffering}
\end{keybox}

\textbf{Google:} ``DASH adaptive streaming over HTTP diagram''\\
\textit{How DASH works: server stores multiple quality versions of each chunk, client dynamically selects bit rate based on bandwidth.}
\end{imagebox}

\subsection{CDN -- Content Distribution Networks}

\begin{warnbox}[Why not one mega-server?]
\begin{itemize}[noitemsep]
    \item Single point of failure
    \item Traffic bottleneck (congestion)
    \item Long distance to remote clients
    \item Multiple copies of video over the same link
    \item \textbf{The solution does not scale!}
\end{itemize}
\end{warnbox}

\begin{keybox}[CDN -- Two strategies]
\textbf{``Enter Deep'':}
\begin{itemize}[noitemsep]
    \item Place CDN servers \textbf{deep inside} access networks, close to users
    \item Many small server clusters
\end{itemize}

\textbf{``Bring Home'':}
\begin{itemize}[noitemsep]
    \item Fewer, \textbf{larger} CDN clusters near access networks (not inside them)
\end{itemize}

The client gets the manifest from the CDN, fetches chunks from the \textbf{best suited server} at the appropriate bit rate.
\end{keybox}

\section{Socket Programming (2.7)}

\subsection{Socket Basics}

\begin{keybox}[Socket = The ``Door'' between Application and Transport]
Two types of sockets for two transport services:
\begin{itemize}[noitemsep]
    \item \textbf{UDP socket}: unreliable datagram (\texttt{SOCK\_DGRAM})
    \item \textbf{TCP socket}: reliable, byte stream-oriented (\texttt{SOCK\_STREAM})
\end{itemize}
\end{keybox}

\subsection{Socket Programming with UDP}

\begin{keybox}[UDP -- No connection]
\begin{itemize}[noitemsep]
    \item \textbf{No handshaking} before data transfer
    \item Sender attaches \textbf{IP address and port number} to each packet
    \item Receiver retrieves sender's IP and port from the packet
    \item Data can be \textbf{lost} or arrive in \textbf{wrong order}
\end{itemize}
\end{keybox}

\textbf{Python UDP client:}
\begin{lstlisting}
from socket import *
serverName = 'hostname'
serverPort = 12000
clientSocket = socket(AF_INET, SOCK_DGRAM)
message = input('Input lowercase sentence:')
clientSocket.sendto(message.encode(), (serverName, serverPort))
modifiedMessage, serverAddress = clientSocket.recvfrom(2048)
print(modifiedMessage.decode())
clientSocket.close()
\end{lstlisting}

\textbf{Python UDP server:}
\begin{lstlisting}
from socket import *
serverPort = 12000
serverSocket = socket(AF_INET, SOCK_DGRAM)
serverSocket.bind(('', serverPort))
print("The server is ready to receive")
while True:
    message, clientAddress = serverSocket.recvfrom(2048)
    modifiedMessage = message.decode().upper()
    serverSocket.sendto(modifiedMessage.encode(), clientAddress)
\end{lstlisting}

\subsection{Socket Programming with TCP}

\begin{keybox}[TCP -- Connection-oriented]
\begin{itemize}[noitemsep]
    \item The server must run first and have created a \textbf{``welcoming socket''}
    \item The client creates a TCP socket with the server's IP + port $\rightarrow$ \textbf{TCP connection} is established
    \item The server creates a \textbf{new socket} (\texttt{connectionSocket}) for each client
    \item Uses \textbf{source port} to distinguish clients
    \item TCP provides reliable, ordered \textbf{byte-stream} transfer
\end{itemize}

\textbf{Python TCP client:}
\begin{lstlisting}
from socket import *
serverName = 'servername'
serverPort = 12000
clientSocket = socket(AF_INET, SOCK_STREAM)
clientSocket.connect((serverName, serverPort))
sentence = input('Input lowercase sentence:')
clientSocket.send(sentence.encode())
modifiedSentence = clientSocket.recv(1024)
print('From Server:', modifiedSentence.decode())
clientSocket.close()
\end{lstlisting}

\textbf{Python TCP server:}
\begin{lstlisting}
from socket import *
serverPort = 12000
serverSocket = socket(AF_INET, SOCK_STREAM)
serverSocket.bind(('', serverPort))
serverSocket.listen(1)
print('The server is ready to receive')
while True:
    connectionSocket, addr = serverSocket.accept()
    sentence = connectionSocket.recv(1024).decode()
    capitalizedSentence = sentence.upper()
    connectionSocket.send(capitalizedSentence.encode())
    connectionSocket.close()
\end{lstlisting}

\subsection{Key Differences: UDP vs. TCP Sockets}

\begin{keybox}[Comparison UDP vs. TCP]
\begin{tabular}{lll}

\midrule
